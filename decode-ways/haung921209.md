## DFS(Memory Limit Exceeded)
```cpp
class Solution {
int cnt;
public:
    bool canBeDecoded(vector<char> st){
        if(st.size()>2){
            return false;
        }else if(st.size()==1){
            return st[0]!='0';
        }
        //st.size()==2

        if(st[0]=='1'){
            return true;
        }else if(st[0]=='2'){
           return (int)(st[1]-'0') <= 6;
        }else{
            return false;
        }
    }

    void helper(string s, int idx){
        if(idx==s.size()){
            cnt++;
            return;
        }else if(idx>s.size()){
            return;
        }
        if(canBeDecoded(vector<char>(s.begin()+idx, s.begin()+idx+1))){
            helper(s, idx+1);
        }

        if(idx+2<=s.size() && canBeDecoded(vector<char>(s.begin()+idx, s.begin()+idx+2))){
            helper(s, idx+2);
        }
    }

    int numDecodings(string s) {
        cnt = 0;
        helper(s, 0);
        return cnt;        
    }
};
```

## DP

```cpp
class Solution {
public:
    int numDecodings(string s) {
        int n = s.size();
        vector<int> dp(n + 1, -1);  // dp[i] = number of ways to decode from index i

        return helper(s, 0, dp);
    }

    int helper(string& s, int idx, vector<int>& dp) {
        if (idx == s.size()) {
            return 1;       
        }
        if (s[idx] == '0') {
            return 0;        
        }
        if (dp[idx] != -1) {
            return dp[idx];   
        }

        int res = helper(s, idx + 1, dp);   

        if (idx + 1 < s.size()) {
            int num = stoi(s.substr(idx, 2));
            if (num >= 10 && num <= 26) {
                res += helper(s, idx + 2, dp); 
            }
        }

        return dp[idx] = res;
    }
};
```

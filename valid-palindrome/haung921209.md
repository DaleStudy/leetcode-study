```cpp
class Solution {
public:
    bool isPalindrome(string s) {
        vector<char> pureAlphabat;
        for(auto cur: s){
            if(cur >= 'a' && cur <='z'){
                pureAlphabat.push_back(cur);
            }else if(cur>= 'A' && cur <='Z'){
                pureAlphabat.push_back(cur-'A'+'a');
            }else if(cur>='0' && cur<='9'){
                pureAlphabat.push_back(cur);
            }
        }
        for(int i=0;i<pureAlphabat.size();i++){
            if(pureAlphabat[i] != pureAlphabat[pureAlphabat.size()-1-i]){
                return false;
            }
        }
        return true;
  }
};
```
